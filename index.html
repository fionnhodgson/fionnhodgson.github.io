Let - declares a variable:
let user = 'John';
let age = 25;
let message = 'Hello';

Const - a variable that never changes:
const myBirthday = '20.01.1992';
const COLOR_ORANGE = "#FF7F00";
// ...when we need to pick a color
let color = COLOR_ORANGE;
alert(color); // #FF7F00

let admin, name; // can declare two variables at once
name = "John";
admin = name;
alert( admin ); // "John"

// Always Use Strict:
"use strict";
// Must be located at the top of a script, excluding comments.

// embed an expression with backticks
alert( `the result is ${1 + 2}` ); // the result is 3
let name = "John";
// embed a variable
alert( `Hello, ${name}!` ); // Hello, John!
